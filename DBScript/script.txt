CREATE TABLE pipe_classes (
    pipe_class_id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    diameter INTEGER NOT NULL,
    working_pressure INTEGER NOT NULL,
    material VARCHAR(100) NOT NULL,
    laying_year INTEGER NOT NULL
);

CREATE TABLE compressor_stations (
    station_id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL UNIQUE,
    location VARCHAR(255) NOT NULL,
    shift_supervisor VARCHAR(100) NOT NULL,
    phone VARCHAR(50) NOT NULL
);

CREATE TABLE pipe_segments (
    segment_id SERIAL PRIMARY KEY,
    pipe_class_id INT NOT NULL,
    segment_number VARCHAR(50) NOT NULL,
    status VARCHAR(50) NOT NULL,
    last_inspection_date DATE NOT NULL,
    length_km DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (pipe_class_id) REFERENCES pipe_classes(pipe_class_id) ON DELETE CASCADE
);

CREATE TABLE pressure_measurements (
    measurement_id SERIAL PRIMARY KEY,
    segment_id INT NOT NULL,
    station_id INT NOT NULL,
    measurement_time TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    pressure DECIMAL(8,2) NOT NULL,
    temperature DECIMAL(5,2) NOT NULL,
    FOREIGN KEY (segment_id) REFERENCES pipe_segments(segment_id) ON DELETE CASCADE,
    FOREIGN KEY (station_id) REFERENCES compressor_stations(station_id) ON DELETE CASCADE
);


CREATE INDEX idx_segments_class ON pipe_segments(pipe_class_id);
CREATE INDEX idx_measurements_segment ON pressure_measurements(segment_id);
CREATE INDEX idx_measurements_station ON pressure_measurements(station_id);
CREATE INDEX idx_measurements_time ON pressure_measurements(measurement_time);
CREATE INDEX idx_stations_name ON compressor_stations(name);
CREATE INDEX idx_pipe_classes_name ON pipe_classes(name);
